interface Printable {
    method printMessage();
}

class Message implements Printable {
    method printMessage(){
        print("Mensagem impressa!");
    }
}

var mensagem = new Message();
mensagem.printMessage();

interface Calculator {
    method add(a: number, b: number): number;
    method subtract(a: number, b: number): number;
}

class Arithmetic implements Calculator {
    method add(a: number, b: number): number {
        return a + b;
    }

    method subtract(a: number, b: number): number {
        return a - b;
    }
}

var aritmetica = new Arithmetic();
aritmetica.add(2,2);
aritmetica.subtract(2,2);
